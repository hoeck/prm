\subsection{Meilensteinplan}
\paragraph*{}
\begin{enumerate}
\item \textbf{M1} Beschaffung Hardware, Schnittstellendefinition (eine Woche)
	\begin{itemize}
	\item Beschaffung und Test Hardware
	\item Beschaffung Möbel
    \item Definition der Schnittstelle Notes <-> Ausleihstation
    \item \textbf{Ergebnisse}
        \begin{itemize}
        \item Funktionierende Hardware
        \item Fertige Schnittstellendefinition
	\end{itemize}
	\end{itemize}
\item \textbf{M2} Ausleihstation (zwei Wochen)
	\begin{itemize}
	\item Testaufbau
    \item Software
    \item \textbf{Ergebnisse} 
    \begin{itemize}
        \item Funktionierende Ausleihstation
    \end{itemize}
	\end{itemize}
\item \textbf{M3} Lotus Notes Schnittstelle (vier Wochen)
	\begin{itemize}
	\item Implementierung und Test
    \item \textbf{Ergebnisse} 
    \begin{itemize}
        \item Fertige Notes Schnittstelle
	\end{itemize}
	\end{itemize}
\item \textbf{M4} Einbau der Hardware (eine Woche)
	\begin{itemize}
	\item Einbau Kartenleser
	\item Einbau Türschloss
	\item Einbau Kamera
	\item Einbau Laser-Barcodescanner
	\item Aufbau Möbel
    \item \textbf{Ergebnisse}
    \begin{itemize}
        \item Fertig im Bibliotheksraum aufgebaute Ausleihstation
        \item Funktionierende Türschlossanlage
	\end{itemize}    
	\end{itemize}
\item \textbf{M5} Endabnahme(eine Woche)
	\begin{itemize}
	\item Finaler Test
	\item Abnahme
    \item \textbf{Ergebnisse}
    \begin{itemize}
        \item Funktionierendes Ausleihsystem
	\end{itemize}
	\end{itemize}
\end{enumerate}
