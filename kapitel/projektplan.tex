\item \textbf{M2} Ausleihstation (drei Wochen)
	\begin{itemize}
	\item Testaufbau

\item \textbf{M3} Lotus Notes Schnittstelle (vier Wochen)
	\begin{itemize}
	\item Implementierung & Test
    \item \textbf{Ergebnisse} 
	\end{itemize}
\item \textbf{M4} Einbau der Hardware (eine Woche)
	\begin{itemize}
	\item Einbau Kartenleser
	\item Einbau Türschloss
	\item Einbau Kamera
	\item Einbau Laser-Barcodescanner
	\item Aufbau Möbel
    \item \textbf{Ergebnisse}
	\end{itemize}
\item \textbf{M5} Endabnahme(eine Woche)
	\begin{itemize}
	\item Finaler Test
	\item Abnahme

	\end{itemize}
